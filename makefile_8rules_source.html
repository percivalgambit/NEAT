12 # function that takes in a list of variables and produces a filtered list of only
13 # variables that have a non-empty expansion
14 filter-defined = $(foreach var,$(1),$(if $($(var)),$(var) ))
15 
16 # list of variables that are also flags to the C++ preprocessor for ftrace, if defined
17 FTRACE_CPP_MACRO_VARS = REPLACE_FP_FN REPLACEMENT_TYPE_FN START_CALLBACK EXIT_CALLBACK
18 
19 # list of variables that are dependencies for the ftrace tool, if defined
20 FTRACE_TOOL_SOURCE_VARS = REPLACE_FP_FILE REPLACEMENT_TYPE_FILE START_CALLBACK_FILE \
21  EXIT_CALLBACK_FILE EXTRA_SOURCE_FILES
22 
23 # filter out undefined variables from FTRACE_CPP_MACRO_VARS and FTRACE_TOOL_DEPS
24 FTRACE_FILTERED_CPP_MACRO_VARS = $(call filter-defined,$(FTRACE_CPP_MACRO_VARS))
25 FTRACE_FILTERED_TOOL_SOURCE_VARS = $(call filter-defined,$(FTRACE_TOOL_SOURCE_VARS))
26 
27 FTRACE_TOOL_SOURCE_DEPS = $(sort $(foreach var,$(FTRACE_TOOL_SOURCE_VARS),$($(var))))
28 
29 # create a list of space-separated flags to pass to the C++ preprocessor that will
30 # define macros to be used in the ftrace source files
31 FTRACE_CPP_FLAGS = $(foreach var,$(FTRACE_FILTERED_CPP_MACRO_VARS),-D$(var)=$($(var)) )
32 
33 # create a list of the object files that will be compiled into the ftrace tool
34 # and that the rule to build the tool will depend on
35 FTRACE_TOOL_OBJ_DEPS = $(sort $(foreach var,\
36  $(FTRACE_FILTERED_TOOL_SOURCE_VARS),\
37  $(patsubst %.cpp,\
38  $(OBJDIR)%$(OBJ_SUFFIX),\
39  $(notdir $($(var)))) ))

72 $(OBJDIR)ftrace$(PINTOOL_SUFFIX): $(OBJDIR)ftrace$(OBJ_SUFFIX) $(FTRACE_TOOL_OBJ_DEPS)
73  $(LINKER) $(TOOL_LDFLAGS) $(LINK_EXE)$@ $(filter %.o,$^) $(TOOL_LPATHS) $(TOOL_LIBS)
74 
75 # since all object files are built in the same directory, but source files can
76 # come from anywhere, we need to programmatically generate rules for every
77 # user-specified object that we need to build
78 define make-ftrace-obj-dep
79 $$(patsubst %.cpp,$$(OBJDIR)%$$(OBJ_SUFFIX),$$(notdir $(1))): $(1)
80  $$(CXX) $$(FTRACE_CPP_FLAGS) $$(TOOL_CXXFLAGS) -I. -I$$(OBJDIR) $$(COMP_OBJ)$$@ $$<
81 endef
82 
83 # generate a rule to build an object from every user-specified source file
84 $(foreach src,$(FTRACE_TOOL_SOURCE_DEPS),$(eval $(call make-ftrace-obj-dep,$(src))))
