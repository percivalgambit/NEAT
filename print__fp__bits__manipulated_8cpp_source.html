28 VOID CountFpMantissaBits(FLT32 flt) {
29  INT32 bits = *reinterpret_cast<INT32 *>(&flt);
30  // Zero out all bits before the mantissa of the floating-point number.
31  bits &= 0x007fffff;
32  // We use the ffs function to find the least significant bit set in the
33  // mantissa, and we need to subtract this number from 24 because it is
34  // 1-indexed.
35  if (bits != 0) {
36  fp_bits_manipulated += 24 - ffs(bits);
37  }
38 }
