36 VOID CountFpMantissaBits(FLT32 flt) {
37  INT32 bits = *reinterpret_cast<INT32 *>(&flt);
38  // Zero out all bits before the mantissa of the floating-point number.
39  bits &= 0x007fffff;
40  // We use the ffs function to find the least significant bit set in the
41  // mantissa, and we need to subtract this number from 24 because it is
42  // 1-indexed.
43  if (bits != 0) {
44  PIN_MutexLock(&fp_bits_manipulated_lock);
45  fp_bits_manipulated += 24 - ffs(bits);
46  PIN_MutexUnlock(&fp_bits_manipulated_lock);
47  }
