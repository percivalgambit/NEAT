211 VOID Routine(RTN rtn, VOID *v) {
212  RTN_Open(rtn);
213 
214  // Forward pass over all instructions in routine
215  for(INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
216 
217  // Increment the count of the total number of instructions
218  INS_InsertCall(ins,
219  IPOINT_BEFORE,
220  (AFUNPTR)docount,
221  IARG_PTR,
222  &ins_count,
223  IARG_END);
224 
225  if (isFpInstruction(ins)) {
226 
227  // Increment the count of floating point instructions
228  INS_InsertCall(ins,
229  IPOINT_BEFORE,
230  (AFUNPTR)docount,
231  IARG_PTR,
232  &fp_count,
233  IARG_END);
234 
235  // If we are in a routine that we have not chosen to instrument, just
236  // record the count of instructions and floating point instructions
237  if (!filter.SelectRtn(rtn))
238  continue;
239 
240 #ifdef REPLACE_FP_FN
241  if (KnobReplaceFPIns)
242  INS_Delete(ins);
243 #endif
244 
245  REGSET regsIn, regsOut;
246  REGSET_Clear(regsIn);
247  REGSET_Clear(regsOut);
248  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 0)));
249  REGSET_Insert(regsOut, REG(INS_OperandReg(ins, 0)));
250 
251  // Increment the count of instrumented floating point instructions
252  INS_InsertCall(ins,
253  IPOINT_BEFORE,
254  (AFUNPTR)docount,
255  IARG_PTR,
256  &instrumented_fp_count,
257  IARG_END);
258 
259  if (INS_OperandIsReg(ins, 1)) {
260  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 1)));
261 
262  // If an instruction is a floating-point instruction that operates on two
263  // registers, call print_reg_fargs and pass it the two operands
264  INS_InsertCall(ins,
265  IPOINT_BEFORE,
266  (AFUNPTR)print_reg_fargs,
267  IARG_UINT32,
268  INS_Opcode(ins),
269  IARG_UINT32,
270  INS_OperandReg(ins, 0),
271  IARG_UINT32,
272  INS_OperandReg(ins, 1),
273  IARG_PARTIAL_CONTEXT,
274  &regsIn,
275  &regsOut,
276  IARG_END);
277  }
278  else {
279  // If an instruction is a floating-point instruction that operates on a
280  // register and a memory location, call print_mem_fargs and pass it the
281  // two operands
282  INS_InsertCall(ins,
283  IPOINT_BEFORE,
284  (AFUNPTR)print_mem_fargs,
285  IARG_UINT32,
286  INS_Opcode(ins),
287  IARG_UINT32,
288  INS_OperandReg(ins, 0),
289  IARG_MEMORYREAD_EA,
290  IARG_PARTIAL_CONTEXT,
291  &regsIn,
292  &regsOut,
293  IARG_END);
294  }
295 
296 
297  // Call print_fresult after every floating-point instruction, and pass it the
298  // result of the instruction
299  INS_InsertCall(ins,
300  IPOINT_AFTER,
301  (AFUNPTR)print_fresult,
302  IARG_UINT32,
303  INS_OperandReg(ins, 0),
304  IARG_CONST_CONTEXT,
305  IARG_END);
306  }
307  }
308

310 }
