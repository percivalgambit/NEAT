220 VOID Routine(RTN rtn, VOID *v) {
221  RTN_Open(rtn);
222 
223  // Forward pass over all instructions in routine
224  for(INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
225 
226  // Increment the count of the total number of instructions
227  INS_InsertCall(ins,
228  IPOINT_BEFORE,
229  (AFUNPTR)docount,
230  IARG_PTR,
231  &ins_count,
232  IARG_END);
233 
234  if (isFpInstruction(ins)) {
235 
236  // Increment the count of floating point instructions
237  INS_InsertCall(ins,
238  IPOINT_BEFORE,
239  (AFUNPTR)docount,
240  IARG_PTR,
241  &fp_count,
242  IARG_END);
243 
244  // If we are in a routine that we have not chosen to instrument, just
245  // record the count of instructions and floating point instructions
246  if (!filter.SelectRtn(rtn))
247  continue;
248 
249 #ifdef REPLACE_FP_FN
250  if (KnobReplaceFPIns)
251  INS_Delete(ins);
252 #endif
253 
254  REGSET regsIn, regsOut;
255  REGSET_Clear(regsIn);
256  REGSET_Clear(regsOut);
257  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 0)));
258  REGSET_Insert(regsOut, REG(INS_OperandReg(ins, 0)));
259 
260  // Increment the count of instrumented floating point instructions
261  INS_InsertCall(ins,
262  IPOINT_BEFORE,
263  (AFUNPTR)docount,
264  IARG_PTR,
265  &instrumented_fp_count,
266  IARG_END);
267 
268  if (INS_OperandIsReg(ins, 1)) {
269  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 1)));
270 
271  // If an instruction is a floating-point instruction that operates on two
272  // registers, call print_reg_fargs and pass it the two operands
273  INS_InsertCall(ins,
274  IPOINT_BEFORE,
275  (AFUNPTR)print_reg_fargs,
276  IARG_UINT32,
277  INS_Opcode(ins),
278  IARG_UINT32,
279  INS_OperandReg(ins, 0),
280  IARG_UINT32,
281  INS_OperandReg(ins, 1),
282  IARG_PARTIAL_CONTEXT,
283  &regsIn,
284  &regsOut,
285  IARG_END);
286  }
287  else {
288  // If an instruction is a floating-point instruction that operates on a
289  // register and a memory location, call print_mem_fargs and pass it the
290  // two operands
291  INS_InsertCall(ins,
292  IPOINT_BEFORE,
293  (AFUNPTR)print_mem_fargs,
294  IARG_UINT32,
295  INS_Opcode(ins),
296  IARG_UINT32,
297  INS_OperandReg(ins, 0),
298  IARG_MEMORYREAD_EA,
299  IARG_PARTIAL_CONTEXT,
300  &regsIn,
301  &regsOut,
302  IARG_END);
303  }
304 
305 
306  // Call print_fresult after every floating-point instruction, and pass it the
307  // result of the instruction
308  INS_InsertCall(ins,
309  IPOINT_AFTER,
310  (AFUNPTR)print_fresult,
311  IARG_UINT32,
312  INS_OperandReg(ins, 0),
313  IARG_CONST_CONTEXT,
314  IARG_END);
315  }
316  }
