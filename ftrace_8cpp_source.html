189 VOID Trace(INS ins, VOID *v) {
190  if (isFpInstruction(ins)) {
191 
192 #ifdef REPLACE_FP_FN
193  if (KnobReplaceFPIns)
194  INS_Delete(ins);
195 #endif
196 
197  REGSET regsIn, regsOut;
198  REGSET_Clear(regsIn);
199  REGSET_Clear(regsOut);
200  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 0)));
201  REGSET_Insert(regsOut, REG(INS_OperandReg(ins, 0)));
202 
203  if (INS_OperandIsReg(ins, 1)) {
204  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 1)));
205 
206  // If an instruction is a floating-point instruction that operates on two
207  // registers, call print_reg_fargs and pass it the two operands
208  INS_InsertCall(ins,
209  IPOINT_BEFORE,
210  (AFUNPTR)print_reg_fargs,
211  IARG_UINT32,
212  INS_Opcode(ins),
213  IARG_UINT32,
214  INS_OperandReg(ins, 0),
215  IARG_UINT32,
216  INS_OperandReg(ins, 1),
217  IARG_PARTIAL_CONTEXT,
218  &regsIn,
219  &regsOut,
220  IARG_END);
221  }
222  else {
223  // If an instruction is a floating-point instruction that operates on a
224  // register and a memory location, call print_mem_fargs and pass it the
225  // two operands
226  INS_InsertCall(ins,

228  (AFUNPTR)print_mem_fargs,
