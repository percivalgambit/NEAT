243 VOID Routine(RTN rtn, VOID *v) {
244  RTN_Open(rtn);
245 
246 #ifdef FUNCTION_LEVEL_REPLACEMENT_TYPE
247  // Push function-level replacement type on stack
248  RTN_InsertCall(rtn,
249  IPOINT_BEFORE,
250  (AFUNPTR)push_function_level_replacement_type,
251  IARG_UINT32,
252  function_replacement_type_map(RTN_Name(rtn)),
253  IARG_END);
254 #endif
255 
256  // Forward pass over all instructions in routine
257  for(INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
258 
259  // Increment the count of the total number of instructions
260  INS_InsertCall(ins,
261  IPOINT_BEFORE,
262  (AFUNPTR)docount,
263  IARG_PTR,
264  &ins_count,
265  IARG_END);
266 
267  if (isFpInstruction(ins)) {
268  INS_Delete(ins);
269 
270  // Set up regsets to create the partial contexts that allow the
271  // instrumentation functions to edit the register values when
272  // replacing floating-point function
273  REGSET regsIn, regsOut;
274  REGSET_Clear(regsIn);
275  REGSET_Clear(regsOut);
276  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 0)));
277  REGSET_Insert(regsOut, REG(INS_OperandReg(ins, 0)));
278 
279  if (INS_OperandIsReg(ins, 1)) {
280  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 1)));
281 
282  // If an instruction is a floating-point instruction that operates
283  // on two registers, call replace_reg_fargs and pass it the two
284  // operands
285  INS_InsertCall(ins,
286  IPOINT_BEFORE,
287  (AFUNPTR)replace_reg_fp_ins,
288  IARG_UINT32,
289  INS_Opcode(ins),
290  IARG_UINT32,
291  INS_OperandReg(ins, 0),
292  IARG_UINT32,
293  INS_OperandReg(ins, 1),
294  IARG_PARTIAL_CONTEXT,
295  &regsIn,
296  &regsOut,
297  IARG_END);
298  }
299  else {
300  // If an instruction is a floating-point instruction that operates
301  // on a register and a memory location, call replace_mem_fargs and
302  // pass it the two operands
303  INS_InsertCall(ins,
304  IPOINT_BEFORE,
305  (AFUNPTR)replace_mem_fp_ins,
306  IARG_UINT32,
307  INS_Opcode(ins),
308  IARG_UINT32,
309  INS_OperandReg(ins, 0),
310  IARG_MEMORYREAD_EA,
311  IARG_PARTIAL_CONTEXT,
312  &regsIn,
313  &regsOut,
314  IARG_END);
315  }
316  }
317  }
