132 VOID Trace(INS ins, VOID *v) {
133  if (isFpInstruction(ins)) {
134 
135  if (INS_OperandIsReg(ins, 1)) {
136  // If an instruction is a floating-point instruction that operates on two
137  // registers, call print_reg_fargs and pass it the two operands
138  INS_InsertCall(ins,
139  IPOINT_BEFORE,
140  AFUNPTR(print_reg_fargs),
141  IARG_UINT32,
142  INS_Opcode(ins),
143  IARG_UINT32,
144  REG(INS_OperandReg(ins, 0)),
145  IARG_UINT32,
146  REG(INS_OperandReg(ins, 1)),
147  IARG_CONTEXT,
148  IARG_END);
149  }
150  else {
151  // If an instruction is a floating-point instruction that operates on a
152  // register and a memory location, call print_reg_fargs and pass it the
153  // two operands
154  INS_InsertCall(ins,
155  IPOINT_BEFORE,
156  AFUNPTR(print_mem_fargs),
157  IARG_UINT32,
158  INS_Opcode(ins),
159  IARG_UINT32,
160  REG(INS_OperandReg(ins, 0)),
161  IARG_MEMORYREAD_EA,
162  IARG_CONTEXT,
163  IARG_END);
164  }
165 
166  // Call print_fresult after every floating-point instruction, and pass it the
167  // result of the instruction

171  IARG_UINT32,
172  REG(INS_OperandReg(ins, 0)),
173  IARG_CONTEXT,
