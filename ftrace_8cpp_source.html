177 VOID Trace(INS ins, VOID *v) {
178  if (isFpInstruction(ins)) {
179 
180 #ifdef REPLACE_FP_FN
181  if (KnobReplaceFPIns)
182  INS_Delete(ins);
183 #endif
184 
185  REGSET regsIn, regsOut;
186  REGSET_Clear(regsIn);
187  REGSET_Clear(regsOut);
188  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 0)));
189  REGSET_Insert(regsOut, REG(INS_OperandReg(ins, 0)));
190 
191  if (INS_OperandIsReg(ins, 1)) {
192  REGSET_Insert(regsIn, REG(INS_OperandReg(ins, 1)));
193 
194  // If an instruction is a floating-point instruction that operates on two
195  // registers, call print_reg_fargs and pass it the two operands
196  INS_InsertCall(ins,
197  IPOINT_BEFORE,
198  (AFUNPTR)print_reg_fargs,
199  IARG_UINT32,
200  INS_Opcode(ins),
201  IARG_UINT32,
202  INS_OperandReg(ins, 0),
203  IARG_UINT32,
204  INS_OperandReg(ins, 1),
205  IARG_PARTIAL_CONTEXT,
206  &regsIn,
207  &regsOut,
208  IARG_END);
209  }
210  else {
211  // If an instruction is a floating-point instruction that operates on a
212  // register and a memory location, call print_mem_fargs and pass it the
213  // two operands
214  INS_InsertCall(ins,
215  IPOINT_BEFORE,
216  (AFUNPTR)print_mem_fargs,
217  IARG_UINT32,
218  INS_Opcode(ins),
219  IARG_UINT32,
220  INS_OperandReg(ins, 0),
221  IARG_MEMORYREAD_EA,
222  IARG_PARTIAL_CONTEXT,
223  &regsIn,
224  &regsOut,
225  IARG_END);
226  }
