140 VOID InstrumentationCallback(const RTN rtn, FpSelector *fp_selector) {
141  RTN_Open(rtn);
142  const string &function_name = RTN_Name(rtn);
143  // clang-format off
144  RTN_InsertCall(
145  rtn, IPOINT_BEFORE,
146  reinterpret_cast<AFUNPTR>(analysis::EnterFunction),
147  IARG_PTR, &function_name,
148  IARG_PTR, fp_selector,
149  IARG_END);
150  // clang-format on
151 
152  // Pass through every instruction in the routine and replace every
153  // floating-point instruction.
154  for (INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
155  if (IsFpInstruction(ins)) {
156  REGSET regs_in, regs_out;
157  REGSET_Clear(regs_in);
158  REGSET_Clear(regs_out);
159  REGSET_Insert(regs_in, INS_OperandReg(ins, 0));
160  REGSET_Insert(regs_out, INS_OperandReg(ins, 0));
161 
162  INS_Delete(ins);
163  if (INS_OperandIsReg(ins, 1)) {
164  REGSET_Insert(regs_in, INS_OperandReg(ins, 1));
165  // clang-format off
166  INS_InsertCall(
167  ins, IPOINT_BEFORE,
168  reinterpret_cast<AFUNPTR>(analysis::ReplaceRegisterFpInstruction),
169  IARG_UINT32, INS_Opcode(ins),
170  IARG_UINT32, INS_OperandReg(ins, 0),
171  IARG_UINT32, INS_OperandReg(ins, 1),
172  IARG_PTR, &function_name,
173  IARG_PTR, fp_selector,
174  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
175  IARG_END);
176  // clang-format on
177  } else {
178  // clang-format off
179  INS_InsertCall(
180  ins, IPOINT_BEFORE,
181  reinterpret_cast<AFUNPTR>(analysis::ReplaceMemoryFpInstruction),
182  IARG_UINT32, INS_Opcode(ins),
183  IARG_UINT32, INS_OperandReg(ins, 0),
184  IARG_MEMORYREAD_EA,
185  IARG_PTR, &function_name,
186  IARG_PTR, fp_selector,
187  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
188  IARG_END);
189  // clang-format on
190  }
191  }
192  }
193  // clang-format off
194  RTN_InsertCall(
195  rtn, IPOINT_AFTER,
196  reinterpret_cast<AFUNPTR>(analysis::ExitFunction),
197  IARG_PTR, &function_name,
198  IARG_PTR, fp_selector,
199  IARG_END);
200  // clang-format on
201  RTN_Close(rtn);
202 }
