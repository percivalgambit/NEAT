131 VOID InstrumentationCallback(const RTN rtn, FpSelector *fp_selector) {
132  RTN_Open(rtn);
133  const string &function_name = RTN_Name(rtn);
134  // clang-format off
135  RTN_InsertCall(
136  rtn, IPOINT_BEFORE,
137  reinterpret_cast<AFUNPTR>(analysis::EnterFunction),
138  IARG_PTR, &function_name,
139  IARG_PTR, fp_selector,
140  IARG_END);
141  // clang-format on
142 
143  // Pass through every instruction in the routine and replace every
144  // floating-point instruction.
145  for (INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
146  if (IsFpInstruction(ins)) {
147  REGSET regs_in, regs_out;
148  REGSET_Clear(regs_in);
149  REGSET_Clear(regs_out);
150  REGSET_Insert(regs_in, INS_OperandReg(ins, 0));
151  REGSET_Insert(regs_out, INS_OperandReg(ins, 0));
152 
153  INS_Delete(ins);
154  if (INS_OperandIsReg(ins, 1)) {
155  REGSET_Insert(regs_in, INS_OperandReg(ins, 1));
156  // clang-format off
157  INS_InsertCall(
158  ins, IPOINT_BEFORE,
159  reinterpret_cast<AFUNPTR>(analysis::ReplaceRegisterFpInstruction),
160  IARG_UINT32, INS_Opcode(ins),
161  IARG_UINT32, INS_OperandReg(ins, 0),
162  IARG_UINT32, INS_OperandReg(ins, 1),
163  IARG_PTR, fp_selector,
164  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
165  IARG_END);
166  // clang-format on
167  } else {
168  // clang-format off
169  INS_InsertCall(
170  ins, IPOINT_BEFORE,
171  reinterpret_cast<AFUNPTR>(analysis::ReplaceMemoryFpInstruction),
172  IARG_UINT32, INS_Opcode(ins),
173  IARG_UINT32, INS_OperandReg(ins, 0),
174  IARG_MEMORYREAD_EA,
175  IARG_PTR, fp_selector,
176  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
177  IARG_END);
178  // clang-format on
179  }
180  }
181  }
