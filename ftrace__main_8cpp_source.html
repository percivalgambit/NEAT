88  // If the KnobFpSelectorName flag is specified on the command line, attempt to
89  // look up the FpSelector from the registry and use it to instrument the

97  ReplaceFpOperations(fp_selector);

100  // If the KnobPrintFpOps flag is specified on the command line, instrument the

102  // formatted as 8 digit hex numbers padded with 0's to a file.
103  const string &print_fp_ops_file_name = KnobPrintFpOps.Value();
104  if (!print_fp_ops_file_name.empty()) {
