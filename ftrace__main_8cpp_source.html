89  // If the KnobFpSelectorName flag is specified on the command line, attempt to
90  // look up the FpSelector from the registry and use it to instrument the
91  // application program with a user-defined FP implementation if it is found.
92  const FpSelectorRegistry *fp_selector_registry =
93  FpSelectorRegistry::GetFpSelectorRegistry();
94  const string &fp_selector_name = KnobFpSelectorName.Value();
95  if (!fp_selector_name.empty()) {
96  FpSelector *fp_selector =
97  fp_selector_registry->GetFpSelectorOrDie(fp_selector_name);
98  ReplaceFpOperations(fp_selector);
99  }
100 
101  // If the KnobPrintFpOps flag is specified on the command line, instrument the
102  // application program to print the arguments and result of every FP operation
103  // formatted as 8 digit hex numbers padded with 0's to a file.
104  const string &print_fp_ops_file_name = KnobPrintFpOps.Value();
105  if (!print_fp_ops_file_name.empty()) {
106  ofstream *print_fp_ops_output =
107  new ofstream(print_fp_ops_file_name.c_str());
108  cerr << "===============================================" << endl
109  << "See file " << print_fp_ops_file_name
110  << " for all FP operations and results" << endl
111  << "===============================================" << endl;
112  PrintFpOperations(print_fp_ops_output);
113  }
114 
115  // If the KnobPrintFpBitsManipulated flag is specified on the command line,

117  // manipulated in every floating-point operation.

119  if (!print_fp_bits_file_name.empty()) {
