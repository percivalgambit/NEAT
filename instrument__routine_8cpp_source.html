40 VOID ReplaceFpOperations(const RTN rtn, FpSelector *fp_selector) {
41  RTN_Open(rtn);
42  const string &function_name = RTN_Name(rtn);
43  // clang-format off
44  RTN_InsertCall(
45  rtn, IPOINT_BEFORE,
46  reinterpret_cast<AFUNPTR>(EnterFunction),
47  IARG_PTR, &function_name,
48  IARG_PTR, fp_selector,
49  IARG_END);
50  // clang-format on
51 
52  // Pass through every instruction in the routine and replace every
53  // floating-point instruction.
54  for (INS ins = RTN_InsHead(rtn); INS_Valid(ins); ins = INS_Next(ins)) {
55  if (IsFpInstruction(ins)) {
56  REGSET regs_in, regs_out;
57  REGSET_Clear(regs_in);
58  REGSET_Clear(regs_out);
59  REGSET_Insert(regs_in, INS_OperandReg(ins, 0));
60  REGSET_Insert(regs_out, INS_OperandReg(ins, 0));
61 
62  INS_Delete(ins);
63  if (INS_OperandIsReg(ins, 1)) {
64  REGSET_Insert(regs_in, INS_OperandReg(ins, 1));
65  // clang-format off
66  INS_InsertCall(
67  ins, IPOINT_BEFORE,
68  reinterpret_cast<AFUNPTR>(ReplaceRegisterFpInstruction),
69  IARG_UINT32, INS_Opcode(ins),
70  IARG_UINT32, INS_OperandReg(ins, 0),
71  IARG_UINT32, INS_OperandReg(ins, 1),
72  IARG_PTR, fp_selector,
73  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
74  IARG_END);
75  // clang-format on
76  } else {
77  // clang-format off
78  INS_InsertCall(
79  ins, IPOINT_BEFORE,
80  reinterpret_cast<AFUNPTR>(ReplaceMemoryFpInstruction),
81  IARG_UINT32, INS_Opcode(ins),
82  IARG_UINT32, INS_OperandReg(ins, 0),
83  IARG_MEMORYREAD_EA,
84  IARG_PTR, fp_selector,
85  IARG_PARTIAL_CONTEXT, &regs_in, &regs_out,
86  IARG_END);
87  // clang-format on
88  }
89  }
90  }
91  // clang-format off
92  RTN_InsertCall(
93  rtn, IPOINT_AFTER,
94  reinterpret_cast<AFUNPTR>(ExitFunction),
95  IARG_PTR, &function_name,
96  IARG_PTR, fp_selector,
97  IARG_END);
98  // clang-format on
99  RTN_Close(rtn);
100 }
